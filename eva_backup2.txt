bool boton() {
  leerBotones();
  return (b1 || b2 || b3);
}

void set_modo(Modo m) {
  modo = m;
  t0 = millis();
  if (m == APAGADO) display.ssd1306_command(SSD1306_DISPLAYOFF);
  else display.ssd1306_command(SSD1306_DISPLAYON);
  display.clearDisplay();
}

void guardar_config_riego() {
  // Inicializar valores
  for (int i = 0; i < 4; i++) {
    riegos[i].dias_desde_ultimo = 0;
    riegos[i].t_planta = i;

    // Calcular dirección base en EEPROM
    int addr = i * sizeof(t_riego);
    EEPROM.put(addr, riegos[i]);
  }
}
void leerRiegos() {
  for (int i = 0; i < 4; i++) {
    int addr = i * sizeof(t_riego);
    EEPROM.get(addr, riegos[i]);
  }
}
void fillEllipse(int16_t x0, int16_t y0, int16_t rx, int16_t ry, uint16_t color) {
  for (int16_t y = -ry; y <= ry; y++) {
    int16_t dx = (int16_t)(rx * sqrt(1.0 - (float)(y * y) / (ry * ry)));
    display.drawFastHLine(x0 - dx, y0 + y, 2 * dx, color);
  }
}

void eva_blink() {
  int i, y, dx, dy, tt;
  for (int dir = 0; dir < 2; dir++) {
    for (i = 0; i <= 10; i++) {
      int k = (dir ? 10 - i : i);
      y = SCREEN_HEIGHT + 20 - (k * (SCREEN_HEIGHT / 2 + 7)) / 10;
      dx = dy = 0;
      if (k > 3 && k < 7) {
        tt = (k - 3) * 3216 / 10;  // seno aproximado *1000
        dx = (3 * tt) / 1000;
        dy = dx / 2;
      }
      display.clearDisplay();
      display.drawBitmap(16 + dx, 17 + dy, eva_eye, 43, 30, SSD1306_WHITE);
      display.drawBitmap(69 + dx, 17 + dy, eva_eye1, 43, 30, SSD1306_WHITE);
      display.fillCircle(SCREEN_WIDTH / 4, y, 20, SSD1306_BLACK);
      display.fillCircle(3 * SCREEN_WIDTH / 4, y, 20, SSD1306_BLACK);
      display.display();
      delay(12);
    }
  }
}

void eva_bounce() {
  randomSeed(analogRead(0));
  myDFPlayer.play(random(7, 10));
  delay(100);
  const int s = 1, d = 30;  // desplazamiento aleatorio y delay

  auto drawEyes = [&](int y) {
    display.clearDisplay();
    int dx = random(-s, s + 1), dy = random(-s, s + 1);
    display.drawBitmap(16 + dx, 17 + dy + y, eva_eye, 43, 30, SSD1306_WHITE);
    display.drawBitmap(69 + dx, 17 + dy + y, eva_eye1, 43, 30, SSD1306_WHITE);
    display.display();
  };

  for (int i = 0; i <= 20; i += 2) { drawEyes(-i); delay(d); }   // subir
  for (int i = 20; i >= 0; i -= 2) { drawEyes(-i); delay(d); }   // bajar
  for (int i = 0; i <= 5; i += 2)  { drawEyes(-i); delay(d); }   // rebote corto
  for (int i = 5; i >= 0; i -= 2)  { drawEyes(-i); delay(d); }
  for (int i = 0; i <= 10; i += 2) { drawEyes(i);  delay(d); }   // bajar final
  for (int i = 10; i >= 0; i -= 2) { drawEyes(i);  delay(d); }
}
void eva_happy() {
  randomSeed(analogRead(0));
  myDFPlayer.play(random(5, 7));
  delay(100);
  const int r = 20, steps = 10, eye_shift = 3;
  const int y_bottom = SCREEN_HEIGHT + r;
  const int y_top = (SCREEN_HEIGHT / 2) + r - 13;
  const int delay_ms = 25;

  auto drawEyes = [&](int dx, int dy, int y) {
    display.clearDisplay();
    display.drawBitmap(16 + dx, 17 + dy, eva_eye, 43, 30, SSD1306_WHITE);
    display.drawBitmap(69 + dx, 17 + dy, eva_eye1, 43, 30, SSD1306_WHITE);
    display.fillCircle(SCREEN_WIDTH / 4, y, r, SSD1306_BLACK);
    display.fillCircle(3 * SCREEN_WIDTH / 4, y, r, SSD1306_BLACK);
    display.display();
  };

  for (int dir = 0; dir < 2; dir++) {  // subir y bajar
    for (int i = 0; i <= steps; i++) {
      int k = dir ? steps - i : i;
      float t = (float)k / steps;
      int y = y_bottom - t * (y_bottom - y_top);

      int dx = 0, dy = 0;
      if (k > steps / 3 && k < 2 * steps / 3) {
        float tt = sin((float)(k - steps / 3) / (steps / 3) * 3.1416);
        dx = (int)(eye_shift * tt);
        dy = dx / 2;
      }

      drawEyes(dx, dy, y);
      delay(delay_ms);

      if (!dir && i == steps) {  // fase de felicidad (1 s)
        unsigned long t0 = millis();
        while (millis() - t0 < 1000) {
          int dxr = random(-6, 7), dyr = random(-6, 7);
          display.clearDisplay();
          display.drawBitmap(16 + dxr, 17 + dyr, eva_eye, 43, 30, SSD1306_WHITE);
          display.drawBitmap(69 + dxr, 17 + dyr, eva_eye1, 43, 30, SSD1306_WHITE);
          display.fillCircle(SCREEN_WIDTH / 4 + dxr, y_top + dyr, r, SSD1306_BLACK);
          display.fillCircle(3 * SCREEN_WIDTH / 4 + dxr, y_top + dyr, r, SSD1306_BLACK);
          display.display();
          delay(60);
        }
      }
    }
  }
}
void printDateTime(const RtcDateTime& dt) {
  char datestring[20];

  snprintf_P(datestring,
             countof(datestring),
             PSTR("%02u/%02u/%04u %02u:%02u:%02u"),
             dt.Month(),
             dt.Day(),
             dt.Year(),
             dt.Hour(),
             dt.Minute(),
             dt.Second());
  Serial.print(datestring);
}
void leerBotones() {
  if(digitalRead(2) == HIGH) b1++;
  else if (digitalRead(2) == LOW) b1=0;
  if(digitalRead(3) == HIGH) b2++;
  else if (digitalRead(3) == LOW) b2=0;
  if(digitalRead(4) == HIGH) b3++;
  else if (digitalRead(4) == LOW) b3=0;
}
void print_menu(int channel) {
  display.clearDisplay();

  display.drawBitmap(50, 20, plant, 30, 30, SSD1306_WHITE);
  display.setTextSize(2);             
  display.setTextColor(SSD1306_WHITE);        
  display.setCursor(0,0);          
  display.print(F("CH"));
  display.print(channel);

   display.setTextSize(2);
  display.setCursor(68, 0);

  if (now.Hour() < 10) display.print('0');
  display.print(now.Hour());
  display.print(':');
  if (now.Minute() < 10) display.print('0');
  display.print(now.Minute());

  display.display();
}
void setup() {
  display.begin(SSD1306_SWITCHCAPVCC, SCREEN_ADDRESS); // iniciamos y limpiamos display
  display.clearDisplay();

  Serial.begin(9600);
  Serial1.begin(9600);

  Serial.print("In.E: ");
  
  if (!myDFPlayer.begin(Serial1)) { //no se detecto el dfplayer
    Serial.println("Err1");
    while (true);
  }

  Serial.print("DFd"); //configuramos el dfplayer
  myDFPlayer.volume(30);
  myDFPlayer.EQ(DFPLAYER_EQ_NORMAL);
  myDFPlayer.outputDevice(DFPLAYER_DEVICE_SD);

  pinMode(8, INPUT); //lowside
  pinMode(9, INPUT);
  pinMode(12, INPUT);
  pinMode(13, INPUT); 

  pinMode(2, INPUT); //botones
  pinMode(3, INPUT);
  pinMode(4, INPUT);

  Rtc.Begin(); //iniciamos el DFplayer
  RtcDateTime compiled = RtcDateTime(__DATE__, __TIME__);
  if (!Rtc.IsDateTimeValid()) Rtc.SetDateTime(compiled);
  if (!Rtc.GetIsRunning()) Rtc.SetIsRunning(true);
  now = Rtc.GetDateTime();
  if (now < compiled) Rtc.SetDateTime(compiled);
  eva_wink();
  leerRiegos();
}

void loop() {
  now = Rtc.GetDateTime();
  //printDateTime(now);
  //Serial.println();
  if (!now.IsValid()) Serial.println("Err2");//RTC lost confidence in the DateTime!
  bool b = boton();
  unsigned long t = millis();

  switch (modo) {

    case ANIMACION:
      if (b) set_modo(MENU);
      else {
        if (t - t_anim > 15000) {  // cada 15 s
          eva_eyes();
          switch (random(4)) {
            case 0: eva_blink(); break;
            case 1: eva_frown(); break;
            case 2: eva_bounce(); break;
            case 3: eva_happy(); break;
          }
          t_anim = t;
        }
        if (t - t0 > 60000) set_modo(APAGADO); // 1 min sin tocar
      }
      break;

     case MENU:
      // --- cambio de canal con pulsos cortos
      if (b1 == 1) { ch--; if (ch < 0) ch = 3; t0 = t; }
      if (b2 == 1) { ch++; if (ch > 3) ch = 0; t0 = t; }
      if (b3 == 1) t0 = t;

      // --- detección de presión prolongada del botón 2
      if (digitalRead(3) == HIGH) {
        if (!b2_presionado) { b2_presionado = true; t_b2 = t; }
        else if (t - t_b2 > 2000) {  // 2 s sostenido
          regar_canal(ch);
          b2_presionado = false;
        }
      } else b2_presionado = false;

      print_menu(ch);
      if (t - t0 > 10000) set_modo(ANIMACION);
      break;

    case APAGADO:
      if (b) set_modo(MENU);
      break;
  }
}
